# From exploring the data, we know the genre columns contain booleans that indicate whether
# a movie is of that genre, or not. So, finding the movies with more than one genre is done
# by summing the booleans, and dropping any results with a count of 1 or less.

# Drop the movie id, movie title, and release date fields, and sum the resulting dataframe by
# rows (axis=1). Name the results column 'Genre Count', and convert the result to a new temporary
# working dataframe df.
df = pd.DataFrame((df_Movies.drop(labels=['movie id','movie title','release date'], axis=1).sum(axis=1, numeric_only=True)),columns=['Genre Count'])

# Find the indexes of all rows in the dataframe where Genre Count is less than or equal to one,
# then drop these rows from the dataframe. The result is a dataframe with the
# Genre Count containing the sum of the genres observed for that movie, with all movies
# of only 1 genre (or none, in case there are observations in the dataset with no genre) dropped.
indexNames = df[df['Genre Count'] <= 1].index
df.drop(indexNames,inplace=True)

# Now join the results dataframe to the movie data to produce a result showing the movie id,
# the title of the movie, and the number of genres. Drop the same indexes from the source Movie
# dataframe (to exclude the movies with <= 1 genre) and create a new dataframe with the movie id
# and movie title columns, left joined to the dataframe containing the genre counts.
df_MultiGenreMovies = (pd.DataFrame((df_Movies.drop(indexNames))[['movie id','movie title']])).join(df,how='left')

# Reset the index for the new dataframe

# My approach is to use the Ratings data and first prepare a DataFrameGroupBy
# which is grouped by movie id, then create a DataFrame using the .count()
# and .mean() values, then drop rows with fewer than 100 ratings. This DataFrame
# is then inner merged with the Movies data - this does the filtering in one step

# Create a working copy of the Ratings DataFrame
df = df_Ratings.copy()

# Drop the superfluous columns
df.drop(labels=['user id','timestamp'], axis=1, inplace=True)

# Create a DataFrameGroupBy using groupby on 'movie id', and create a new DataFrame
# populated with the .count() value - gives the number of times the 'movie id' appears
# in the Ratings observations
df_TopRatings = pd.DataFrame(df.groupby('movie id').count())

# Now add the average rating to the new DataFrame, and tidy up the labels
df_TopRatings['average rating'] = df.groupby('movie id').mean()
df_TopRatings.rename(columns = {'rating':'number of ratings'}, inplace=True)

# Find the indexes for where there are fewer than 100 ratings and drop them
indexNames = df_TopRatings[df_TopRatings['number of ratings'] < 100].index
df_TopRatings.drop(indexNames, inplace=True)

# Create a new DataFrame with an inner merge of the original Movies data and the
# new DataFrame which now contains movie id, number of ratings, and average rating
# only for movies with more than 100 ratings. Inner merge on 'movie id' excludes all
# movies with fewer than 100 ratings.
df_TopMovies = pd.merge(df_Movies, df_TopRatings, how='inner', on='movie id')

# Sort the DataFrame by 'average rating', with ascending=False so the sorting is highest
# to lowest average rating
df_TopMovies.sort_values(by='average rating', ascending=False, inplace=True)

# Create the finished DataFrame by extracting the first 25 rows
df_Top25Movies = df_TopMovies.iloc[0:25]
